### 一些疑难点  
1. **Kosaraju算法中，可以直接用后序在原图上DFS，而不建逆图用逆后序DFS嘛 ？**   
这样写HDU和POJ的两道模板题都能过，但是是有问题的！  
反例就是一位大佬在知乎上提出来的：  
1 -> 2, 2 -> 3, 3 -> 4, 4 -> 1, 1 -> 5  
参见链接：[https://www.zhihu.com/question/265266923/answer/329700845](https://www.zhihu.com/question/265266923/answer/329700845)  
2. **Tarjan算法中为什么已访问过的点要写 low[u] = min(low[u], dfn[v]) 而不写 low[u] = min(low[u], dfn[v]) ？**  
这样写HDU和POJ的两道模板题也都能过，而且还没发现什么反例……   
但是 low[u] = min(low[u], dfn[v]) ，代表遇到当前栈中已访问的点v，且v的时间戳比目前u节点记录的最早low[u]小，就更新low[u]，从算法理解层面来讲这样写合理  
3. **点双连通分量**  
点双连通分量并不一定形成环，例如在下面的图中：  
1 - 2, 2 - 3, 2 - 4, 1 - 5, 1 - 6, 5 - 6   
点双连通分量总共是： 1 - 5 - 6 - 1, 1 - 2， 2 - 3, 2 - 4  
割顶是：1, 2  
4. **强连通分量， 边双连通分量**  
二者其实都是一样的，都是形成环  
只不过前者用于有向图，后者用于无向图

### Popular Cows - POJ 2186
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180325%20Problem0325/Popular%20Cows%20-%20POJ%202186.png)
**题目大意**  
在牛群中，A牛喜欢B牛，B牛喜欢C牛，那么A牛喜欢C牛，即喜欢具有传递性，现在问有多少头牛是大家都喜欢的  
<br>
**思路**    
这题作为强连通分量 + 缩点的模板题  
同一个强连通分量内的牛是互相喜欢的，所以缩点处理，缩点后找到出度为0的牛群的数量  
如果这样的牛群只有一个，那么那个牛群的数量就是答案了，如果有多个，那么答案为0，因为至少有两个牛群中的任意一方不喜欢另一方   
```cpp
  //Kosaraju算法解决强连通分量问题
  #include <cstdio>
  #include <iostream>
  #include <algorithm>
  #include <cstring>
  #include <vector>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  typedef long long ll;
  const int N = 50000 + 15;
  const int INF = 0x3f3f3f3f;

  struct edge {int to, next;};
  edge e[N], et[N];
  int head[N], headt[N];
  int tot;
  int stk[N], pstk;                   //栈
  bool used[N];
  int mp[N], out[N], cnt[N];

  inline void scan_d(int& ret){       //输入外挂
      char c;
      ret = 0;
      while((c = getchar()) < '0' || c > '9');
      while('0' <= c && c <= '9'){
          ret = ret*10 + c - '0';
          c = getchar();
      }
  }

  inline void init(){
      met(head, -1);
      met(headt, -1);
      met(mp, 0);
      met(out, 0);
      met(cnt, 0);
      tot = 0, pstk = 0;
  }

  inline void addEdge(int from, int to){
      e[tot].to = to;             //建立原图
      e[tot].next = head[from];
      head[from] = tot;

      et[tot].to = from;          //建立逆图
      et[tot].next = headt[to];
      headt[to] = tot++;
  }

  void dfs0(int u){
      used[u] = true;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          if(!used[v])    dfs0(v);
      }
      stk[++pstk] = u;            //逆后序入栈
  }

  void dfs1(int u, int u0){
      used[u] = true;
      for(int i = headt[u]; ~i; i = et[i].next){
          int v = et[i].to;
          if(!used[v])    dfs1(v, u0);
      }
      mp[u] = u0;                 //缩点
      cnt[u0]++;                  //统计缩点后的数量
  }

  int main(){
      int n, m;
      while(~scanf("%d", &n)){
          init();
          scan_d(m);
          int t = 0, ans = 0;
          while(m--){
              int from, to;
              scan_d(from);
              scan_d(to);
              addEdge(from, to);
          }
          met(used, false);
          for(int i = 1; i <= n; i++){
              if(!used[i])    dfs0(i);
          }

          met(used, false);
          do{
              int u = stk[pstk];      //以逆后序进行第二遍DFS
              if(!used[u])    dfs1(u, u);
          }while(--pstk);

          for(int i = 0; i < tot; i++){
              int u = et[i].to, v = e[i].to;
              if(mp[u] == mp[v])  continue;   //在同一个连通分量的不统计出度
              out[mp[u]]++;
          }

          for(int i = 1; i <= n; i++){
              if(mp[i] != i)  continue;
              if(out[i] == 0){
                  t++;                    //出度为0的点的数量
                  ans = cnt[i];
              }
          }
          if(t == 1){
              printf("%d\n", ans);
          }else{
              printf("0\n");
          }
      }
      return 0;
  }
```

```cpp
  //Tarjan算法解决强连通分量问题
  #include <cstdio>
  #include <algorithm>
  #include <iostream>
  #include <cstring>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  const int N = 50005;

  struct edge { int to, next; };
  edge e[N];
  int head[N];
  int dfn[N], low[N], ts;
  int stk[N], pstk;
  int mp[N], cnt[N], out[N];
  bool instk[N];
  int tot;

  inline void scan_d(int& ret){
      char c;
      ret = 0;
      while((c = getchar()) < '0' || c > '9');
      while('0' <= c && c <= '9'){
          ret = ret*10 + c - '0';
          c = getchar();
      }
  }

  inline void init(){
      met(head, -1);
      met(dfn, 0);
      met(low, 0);
      met(instk, false);
      met(cnt, 0);
      met(out, 0);
      tot = 0, pstk = 0, ts = 1;
  }

  void addEdge(int from, int to){
      e[tot].to = to;
      e[tot].next = head[from];
      head[from] = tot++;
  }

  void tarjan(int u){
      dfn[u] = low[u] = ts++;     //印上时间戳
      stk[++pstk] = u;            //入栈
      instk[u] = true;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          if(!dfn[v]){
              tarjan(v);
              low[u] = min(low[u], low[v]);
          }else if(instk[v]){
              low[u] = min(low[u], dfn[v]);
          }
          if(!instk[v])   out[u]++;       //不在栈中就说明是缩点后指向的别的点，可统计出度
      }
      if(dfn[u] == low[u]){       //回溯找不到比u时间戳更小的点，则栈顶到u为一个强连通分量
          do{
              int v = stk[pstk];
              mp[v] = u;
              out[u] += (u != v) * (out[v]);    //排除掉自身，因为是统计了的
              cnt[u]++;
              instk[u] = false;
          }while(stk[pstk--] != u);
      }
  }

  int main(){
      int n, m;
      while(~scanf("%d", &n)){
          scan_d(m);
          init();
          int t = 0, ans = 0;
          while(m--){
              int from, to;
              scan_d(from);
              scan_d(to);
              addEdge(from, to);
          }
          for(int i = 1; i <= n; i++){
              if(!dfn[i])  tarjan(i);
          }
          for(int i = 1; i <= n; i++){
              if(mp[i] != i)   continue;
              if(out[i] == 0){
                  t++;
                  ans = cnt[i];
              }
          }
          printf("%d\n", (t == 1 ? ans : 0));
      }
      return 0;
  }
```

### Data Center Maintenance - CodeForces-950E
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180325%20Problem0325/Data%20Center%20Maintenance%20-%20CodeForces-950E.jpg)
**题目大意**  
数据中心有n台服务器，他们的每个客户的数据放在两个不同的服务器上，假设一天有h小时。现在数据中心的服务器在0至h - 1时间内需要各自维护一个小时，给定每台服务器维护的时刻表，这个时刻表保证每名客户能在任意时刻访问到自己的数据，现在问将其中至少多少台服务器的维护时刻推后一个小时（其中h - 1时刻推后到0时刻），能使得每名客户都还能在任意时刻访问到自己的数据    
<br>
**思路**    
“ACM的题目具有欺骗性”， 这题乍一看，和强连通分量半毛钱关系都没有呀！  
都是不妨想想，如果放置同一名客户的两台服务器维护时刻相差1,那么其中维护时刻小的那台向后推移，另一台就得跟着向后推移，由此可以构成一个有向图，其中被指向的点是受起点推移而推移的点，在一个圈内的点，一个推移，个个推移，所以可以求强连通分量后缩点，找到出度为0并且数量最小的点，就是答案了  
```cpp
  #include <cstdio>
  #include <algorithm>
  #include <iostream>
  #include <cstring>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  const int N = 200000 + 5;
  const int INF = 0x3f3f3f3f;

  struct edge { int to, next; };
  edge e[N];
  int head[N];
  int tot;

  int mp[N];
  int cnt[N];
  int dfn[N], low[N], out[N];
  int ts;
  int stk[N], pstk;
  bool instk[N];
  int val[N];

  inline void scan_d(int& ret){
      char c;
      ret = 0;
      while((c = getchar()) < '0' || c > '9');
      while('0' <= c && c <= '9'){
          ret = ret*10 + c - '0';
          c = getchar();
      }
  }

  inline void init(){
      met(head, -1);
      met(cnt, 0);
      met(dfn, 0);
      met(low, 0);
      met(instk, false);
      met(out, 0);
      tot = 0, pstk = 0, ts = 1;
  }

  inline void addEdge(int from, int to){
      e[tot].to = to;
      e[tot].next = head[from];
      head[from] = tot++;
  }

  void tarjan(int u){
      if(dfn[u])     return;
      dfn[u] = low[u] = ts++;
      stk[++pstk] = u;
      instk[u] = true;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          if(!dfn[v]){
              tarjan(v);
              low[u] = min(low[u], low[v]);
          }else if(instk[v]){
              low[u] = min(low[u], dfn[v]);
          }
          if(!instk[v])   out[u]++;
      }
      if(dfn[u] == low[u]){
          do{
              int v = stk[pstk];
              cnt[u]++;
              mp[v] = u;
              instk[v] = false;
          }while(stk[pstk--] != u);
      }
  }

  int main(){
      int n, m, h;
      scan_d(n);
      scan_d(m);
      scan_d(h);
      init();
      int ans_cnt = INF, ans;
      for(int i = 1; i <= n; i++)     scan_d(val[i]);
      for(int i = 1; i <= m; i++){
          int a, b;
          scan_d(a);
          scan_d(b);
          if(val[a] == (val[b] + 1)%h)    addEdge(b, a);
          if(val[b] == (val[a] + 1)%h)    addEdge(a, b);
      }
      for(int i = 1; i <= n; i++){
          tarjan(i);
      }
      for(int i = 1; i <= n; i++){
          if(mp[i] == i)  continue;
          out[mp[i]] += out[i];
      }
      for(int i = 1; i <= n; i++){
          if(mp[i] != i)  continue;
          if(out[i] == 0 && ans_cnt > cnt[i]){
              ans_cnt = cnt[i];
              ans = i;
          }
      }

      printf("%d\n", ans_cnt);
      for(int i = 1; i <= n; i++){
          if(mp[i] == ans){
              printf("%d", i);
              if(ans_cnt-- && ans_cnt > 0)    printf(" ");
          }
      }
      printf("\n");

      return 0;
  }
```

### Mr. Kitayuta's Technology - CodeForces-505D  
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180325%20Problem0325/Mr.%20Kitayuta%27s%20Technology%20-%20CodeForces-505D.png)  
**题目大意**  
给定一个图，求最少用几条边重新架构其中的弱连通分量而保证原图各点的连通性依旧存在  
<br>
**思路**    
翻了一下题解……自己宛若智障……  
如果原来那个弱连通分量里有强连通分量，那就干脆把一整个图改为强联通图，否则就改成树，对应的答案为（点数）和（点数 - 1），再求和就行了  
```cpp
  #include <cstdio>
  #include <iostream>
  #include <algorithm>
  #include <cstring>
  #include <queue>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  const int N = 1e5 + 15;
  const int INF = 0x3f3f3f3f;

  struct edge{ int to, next; };
  edge e[N];
  int head[N];
  int tot;

  int low[N], dfn[N];
  int stk[N], pstk;
  int mp[N], cnt[N];
  bool instk[N];
  int ts;

  int ft[N];
  int ufcnt[N];
  bool circle[N];

  void init(){
      met(head, -1);
      met(low, 0);
      met(dfn, 0);
      met(instk, false);
      met(cnt, 0);
      met(ufcnt, 0);
      met(circle, 0);
      for(int i = 1; i <= N; i++){
          ft[i] = i;
      }
      ts = 1, pstk = 0, tot = 0;
  }

  int ufFind(int x) { return ft[x] == x ? x : ft[x] = ufFind(ft[x]); }
  void ufUnion(int x, int y){
      int p = ufFind(x), q = ufFind(y);
      if(p != q)  ft[q] = p;
  }

  void addEdge(int from, int to){
      e[tot].to = to;
      e[tot].next = head[from];
      head[from] = tot++;
  }

  void tarjan(int u){
      if(dfn[u])      return;
      dfn[u] = low[u] = ts++;
      instk[u] = true;
      stk[++pstk] = u;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          ufUnion(u, v);
          if(!dfn[v]){
              tarjan(v);
              low[u] = min(low[u], low[v]);
          }else if(instk[v]){
              low[u] = min(low[u], dfn[v]);
          }
      }
      if(dfn[u] == low[u]){
          do{
              int v = stk[pstk];
              instk[v] = false;
              mp[v] = u;
              cnt[u]++;
          }while(stk[pstk--] != u);
      }
  }

  int main(){
      int n, m;
      while(~scanf("%d%d", &n, &m)){
          int ans = 0;
          init();
          while(m--){
              int from, to;
              scanf("%d%d", &from, &to);
              addEdge(from, to);
          }
          for(int i = 1; i <= n; i++){
              tarjan(i);
          }
          for(int i = 1; i <= n; i++){
              int ift = ufFind(i);
              if(mp[i] != i)      continue;
              if(cnt[i] != 1){
                  circle[ift] = true;
                  ufcnt[ift] += cnt[i] - 1;
              }
              ufcnt[ift]++;
          }
          for(int i = 1; i <= n; i++){
              int ift = ufFind(i);
              if(ift != i)  continue;
              ans += ufcnt[i] - (circle[i] == 0);
          }
          printf("%d\n", ans);
      }
      return 0;
  }
```

### SPF - POJ 1523
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180325%20Problem0325/SPF%20-%20POJ%201523.jpg)
**题目大意**  
求连通图中的割点以及去掉该割点后有多少个连通分量    
<br>
**思路**    
求割点是没什么好说的，问题是求去掉割点后有多少个连通分量  
有点儿考算法理解程度……回溯时 low[v] >= dfn[u]，不仅代表u是割点，也代表v这棵子树是u的一个孩子，那么如果u是根节点，去掉后连通分量数就是u的孩子数（显而易见），如果u不是根节点，去掉后连通分量数则为 （u的孩子数 + 1），这个‘1’是u的父节点以上那一块  
```cpp
  #include <cstdio>
  #include <iostream>
  #include <algorithm>
  #include <cstring>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  const int N = 1000005 + 15;
  const int M = 1000 + 15;
  const int INF = 0x3f3f3f3f;

  struct edge { int to, next; };
  edge e[N];
  int tot;
  int head[M];

  int dfn[M], low[M], cnt[N];
  int ts;
  int stk[N], pstk;
  int node_child[M];
  bool iscut[M];
  bool root[M];
  bool vis[M];
  int ft[M];

  void init(){
      met(head, -1);
      met(dfn, 0);
      met(low, 0);
      met(iscut, 0);
      met(node_child, 0);
      met(root, 0);
      tot = 0, ts = 1, pstk = 0;
  }

  void addEdge(int from, int to){
      e[tot].to = to;
      e[tot].next = head[from];
      head[from] = tot++;
  }

  void tarjan(int u, int pa){
      dfn[u] = low[u] = ts++;
      int child = 0;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          if(v == pa)     continue;
          if(!dfn[v]){
              child++;
              tarjan(v, u);
              low[u] = min(low[u], low[v]);
              if(low[v] >= dfn[u]){
                  node_child[u]++;
                  iscut[u] = true;
              }
          }else if(dfn[v] < dfn[u]){
              low[u] = min(low[u], dfn[v]);
          }
      }
      if(pa == -1 && child == 1)  iscut[u] = 0;
  }

  int main(){
      int a, b;
      int no = 1;
      while(true){
          init();
          b = -1;
          bool flag = true;
          while(scanf("%d", &a) && a){
              scanf("%d", &b);
              addEdge(a, b);
              addEdge(b, a);
          }
          if(b == -1)    break;
          if(no != 1)     printf("\n");
          printf("Network #%d\n", no++);
          for(int i = 1; i <= 1000; i++){
              if(!dfn[i]){
                  root[i] = true;
                  tarjan(i, -1);
              }
          }
          for(int i = 1; i <= 1000; i++){
              if(iscut[i]){
                  flag = false;
                  printf("  SPF node %d leaves %d subnets\n", i, node_child[i] + (root[i] == false));
              }
          }
          if(flag){
              printf("  No SPF nodes\n");
          }
      }
      return 0;
  }
```

### Mining Your Own Business -  UVALive - 5135  
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180325%20Problem0325/5135.png)  
**题目大意**  
假设一开始的一个图是不与外界连通的，在这个图中至少加入多少条连接外界的边，能使得这个图去掉任意点后的其余点通过边仍能和外界连通    
<br>
**思路**    
首先，可以肯定不能加在割顶，因为把割顶去掉会直接导致割顶连接的两个点双连通分量不能连接外界   
第二，一个点双连通中如果有一个割顶，那么这个双连通分量中加一条边就可以了，不加在割顶上  
第三，如果一个点双连通中有多个割顶，那么这个双连通中也不需要加边，因为有多个割顶必定连接多个点双连通分量，任何一个点挂掉，必定仍然和某个双连通相连，而一个割顶的点双连通中又含有一条与外界相连的边  
特殊的，如果一个图中没有割顶，那么选择任意位置的两个点加边即可  
最后一波乘法原理算出有多少种可能性  
```cpp
  #include <cstdio>
  #include <iostream>
  #include <algorithm>
  #include <cstring>
  #include <vector>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  typedef long long ll;
  const int N = 1e5 + 15;
  const int INF = 0x3f3f3f3f;

  struct edge { int from, to, next; };
  edge e[N];
  int tot;
  int head[N];

  int dfn[N], low[N], cnt[N];
  int ts;
  int stk[N], pstk;
  bool iscut[N];
  int bcc_cnt[N], bcc_cut[N];
  int pbcc;

  inline void scan_d(int& ret) {
      register char c;
      ret = 0;
      while ((c = getchar()) < '0' || c > '9');
      while (c >= '0' && c <= '9')
      {
          ret = ret * 10 + (c - '0'), c = getchar();
      }
  }

  void init(){
      met(head, -1);
      met(dfn, 0);
      met(low, 0);
      met(iscut, 0);
      met(bcc_cnt, 0);
      met(bcc_cut, 0);
      tot = 0, ts = 1, pstk = 0, pbcc = 0;
  }

  void addEdge(int from, int to){
      e[tot].from = from;
      e[tot].to = to;
      e[tot].next = head[from];
      head[from] = tot++;
  }

  void tarjan(int u, int pa){
      dfn[u] = low[u] = ts++;
      int child = 0;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          if(v == pa)     continue;
          if(!dfn[v]){
              stk[++pstk] = i;    //入栈的是边
              tarjan(v, u);
              child++;
              low[u] = min(low[u], low[v]);   
              if(low[v] >= dfn[u]){
                  iscut[u] = true;
                  pbcc++;
                  do{
                      int ecur = stk[pstk];
                      bcc_cnt[pbcc]++;      //所属双连通分量点个数++
                      if(iscut[e[ecur].to])  bcc_cut[pbcc]++;   //是割点，则所属双连通分量割点数++
                  }while(stk[pstk--] != i);
                  bcc_cut[pbcc]++;          //还要算上u
                  bcc_cnt[pbcc]++;
              }
          }else if(dfn[v] < dfn[u]){
              low[u] = min(low[u], dfn[v]);
          }
      }
      if(child == 1 && pa == -1){
          iscut[u] = false;
          bcc_cut[pbcc]--;                  //不是割点则对应双连通分量割点数--，根节点必定在最后才计算双连通分量，因为回溯嘛
      }
  }

  int main(){
      int n, caseno = 0;
      while(true){
          scan_d(n);
          if(!n)  break;
          init();
          int upper = 0;
          ll t = 0, ans = 1;
          while(n--){
              int a, b;
              scan_d(a);
              scan_d(b);
              addEdge(a, b);
              addEdge(b, a);
              upper = max(upper, max(a, b));
          }
          for(int i = 1; i <= upper; i++){
              if(dfn[i])  continue;
              pbcc = 0;
              pstk = 0;
              met(bcc_cnt, 0);
              met(bcc_cut, 0);
              tarjan(i, -1);
              if(pbcc == 1){
                  t += 2;
                  ans*=(ll)bcc_cnt[1] * (ll)(bcc_cnt[1] - 1)/2;
              }else{
                  for(int i = 1; i <= pbcc; i++){
                      if(bcc_cut[i] > 1) continue;
                      t++;
                      ans*=(ll)(bcc_cnt[i] - 1);
                  }
              }
          }
          printf("Case %d: %lld %lld\n", ++caseno, t, ans);
      }

      return 0;
  }
```

### Road Construction - POJ-3352   
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180325%20Problem0325/Road%20Construction%20-%20POJ-3352.png)  
**题目大意**  
给定一个图，问加入几条边后能使整个图是边双连通图      
<br>
**思路**    
首先原有双连通分量缩点，缩完后找度数为1的点（也就是树的叶子或者只有一个孩子的根），每两个接一条边就行，如果是奇数的话，只好任选一个叶子加一条边，所以答案是（度数为1的点的个数 + 1）/2  
```cpp
  #include <cstdio>
  #include <iostream>
  #include <algorithm>
  #include <cstring>
  #include <vector>
  #define met(a, b) memset(a, b, sizeof(a))
  using namespace std;
  typedef long long ll;
  const int N = 1001;
  const int INF = 0x3f3f3f3f;

  struct edge { int from, to, next; };
  edge e[N << 1];
  int tot;
  int head[N];

  int dfn[N], low[N], mp[N];
  int ts;
  int stk[N], pstk;
  int indegree[N];

  inline void input(int& ret){
      char c;
      ret = 0;
      while((c = getchar()) < '0' || c > '9');
      while('0' <= c && c <= '9'){
          ret = ret*10 + (c - '0');
          c = getchar();
      }
  }

  void init(){
      met(head, -1);
      met(dfn, 0);
      met(low, 0);
      met(mp, 0);
      met(indegree, 0);
      tot = 0, ts = 1, pstk = 0;
  }

  void addEdge(int from, int to){
      e[tot].to = to;
      e[tot].next = head[from];
      head[from] = tot++;
  }

  void tarjan(int u, int pa){
      dfn[u] = low[u] = ts++;
      stk[++pstk] = u;
      for(int i = head[u]; ~i; i = e[i].next){
          int v = e[i].to;
          if(v == pa)  continue;
          if(!dfn[v]){
              tarjan(v, u);
              low[u] = min(low[u], low[v]);
          }else{
              low[u] = min(low[u], dfn[v]);
          }
      }
      if(dfn[u] == low[u]){
          do{
              int v = stk[pstk];
              mp[v] = u;
          }while(stk[pstk--] != u);
      }
  }

  int main(){
      int n, m;
      input(n);
      input(m);
      int ans = 0;
      init();
      while(m--){
          int a, b;
          input(a);
          input(b);
          addEdge(a, b);
          addEdge(b, a);
      }
      for(int i = 1; i <= n; i++){
          if(!dfn[i])     tarjan(i, -1);
      }
      for(int u = 1; u <= n; u++){
          for(int i = head[u]; ~i; i = e[i].next){
              int v = e[i].to;
              if(mp[v] == mp[u])  continue;
              indegree[mp[v]]++;
          }
      }
      for(int i = 1; i <= n; i++){
          if(mp[i] != i)  continue;
          if(indegree[i] == 1)    ans++;
      }
      printf("%d\n", (ans + 1) >> 1);

      return 0;
  }
```
