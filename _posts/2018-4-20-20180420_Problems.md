（略！）

**思路**    
模板题  
入门打法    
```cpp
  #include <cstring>
  #include <cstdio>
  #include <algorithm>
  #include <set>
  using namespace std;
  const int N = 60000;
  const int ascii_size = 96;

  set<int> st;
  char str[10000 + 1];
  int  total_num;

  int  pos[N];
  int  nxt[N][ascii_size];
  int  fail[N];
  int  tot;
  int  root;

  int  que[N];
  int  head, tail;

  inline void init(){
      memset(pos, 0, sizeof(pos));
      memset(nxt[0], 0, sizeof(nxt));
      memset(fail, 0, sizeof(fail));
      total_num = 0;
      head = tail = 0;
      tot = 1;
      root = tot++;
  }

  void push(char* s, int i){    //此处与Trie建树相同
      int p = root;
      while(* s){
          int idx = * s - 32;   //-32是减掉不可见的ASCII码
          if(!nxt[p][idx]){
              nxt[p][idx] = tot++;
          }
          p = nxt[p][idx];
          s++;
      }
      pos[p] = i;
  }

  void setFail(){             //设置Fail指针，使用BFS
      que[head++] = root;     //root结点进队列
      while(tail != head){
          int p = que[tail++];
          for(int idx = 0; idx < ascii_size; idx++){
              if(nxt[p][idx]){
                  que[head++] = nxt[p][idx];    //存在结点则进队列
                  int q = fail[p];
                  while(q){     //搜索p的失配指针，试图找到可匹配的结点
                      if(nxt[q][idx]){
                          fail[nxt[p][idx]] = nxt[q][idx];
                          break;
                      }
                      q = fail[q];
                  }
                  if(!q)  fail[nxt[p][idx]] = root;   //实在找不到，就连回root
              }
          }
      }
  }

  void query(char * s, int i){
      st.clear();
      int p = root;
      while(*s){
          int idx = * s - 32;
          while(p && nxt[p][idx] == 0)    p = fail[p];    //失配时沿Fail指针不断寻找
          p = (nxt[p][idx] ? nxt[p][idx] : root);         //实在找不到，就回到root结点
          while(pos[p] && p != root){                     //找到了的话则进入循环，所有Fail指针都是匹配的结果
              st.insert(pos[p]);
              p = fail[p];
          }
          s++;
      }
      if(!st.empty()){
          total_num++;
          printf("web %d:", i);
          for(set<int>::iterator it = st.begin(); it != st.end(); it++){
              printf(" %d", * it);
          }
          puts("");
      }
  }

  int main(){
      int n;
      while(~scanf("%d", &n)){
          init();
          for(int i = 1; i <= n; i++){
              scanf("%s", str);
              push(str, i);
          }
          setFail();

          int m;
          scanf("%d", &m);
          for(int i = 1; i <= m; i++){
              scanf("%s", str);
              query(str, i);
          }
          printf("total: %d\n", total_num);
      }
      return 0;
  }
```

比较简约的打法：
```cpp
  #include <cstring>
  #include <cstdio>
  #include <algorithm>
  #include <set>
  using namespace std;
  const int N = 60000;
  const int ascii_size = 96;

  set<int> st;
  char str[10000 + 1];
  int  total_num;

  int  pos[N];
  int  nxt[N][ascii_size];
  int  fail[N];
  int  tot;

  int  que[N];
  int  head, tail;

  inline void init(){
      memset(pos, 0, sizeof(pos));
      memset(nxt[0], -1, sizeof(nxt));
      memset(fail, 0, sizeof(fail));    //fail直接全部指向root节点，root节点为0
      total_num = 0;
      head = tail = 0;
      tot = 1;
  }

  void push(char * s, int i){
      int p = 0;
      while(*s){
          int idx = * s - 32;
          if(nxt[p][idx] == -1){
              nxt[p][idx] = tot++;
          }
          p = nxt[p][idx];
          s++;
      }
      pos[p] = i;
  }

  void setFail(){
      for(int idx = 0; idx < ascii_size; idx++){
          if(nxt[0][idx] != -1){
              que[head++] = nxt[0][idx];
          }else{
              nxt[0][idx] = 0;    //不存在则直接指向root结点，符合匹配的规则
          }
      }
      while(tail != head){
          int p = que[tail++];
          for(int idx = 0; idx < ascii_size; idx++){
              if(~nxt[p][idx]){
                  //存在时设置Fail指针，因为设置过与root直接相连的结点的nxt值，所以不会为-1,不存在时会为0，
                  //直接指向root节点，符合匹配规则
                  fail[nxt[p][idx]] = nxt[fail[p]][idx];    
                  que[head++] = nxt[p][idx];
              }else{
                  nxt[p][idx] = nxt[fail[p]][idx];  //不存在时直接指向nxt[fail[p]][idx]
              }
          }
      }
  }

  void query(char* s, int i){
      int p = 0;
      st.clear();
      for(; *s; s++){
          int idx = * s - 32;
          p = nxt[p][idx];
          for(int q = p; q && pos[q]; q = fail[q]){
              st.insert(pos[q]);
          }
      }
      if(!st.empty()){
          total_num++;
          printf("web %d:", i);
          for(set<int>::iterator it = st.begin(); it != st.end(); it++){
              printf(" %d", * it);
          }
          puts("");
      }
  }

  int main(){
      int n;
      while(~scanf("%d", &n)){
          init();
          for(int i = 1; i <= n; i++){
              scanf("%s", str);
              push(str, i);
          }
          setFail();

          int m;
          scanf("%d", &m);
          for(int i = 1; i <= m; i++){
              scanf("%s", str);
              query(str, i);
          }
          printf("total: %d\n", total_num);
      }
      return 0;
  }
```

### 病毒侵袭持续中 - HDU 3065
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180420%20Problem0420/%E7%97%85%E6%AF%92%E4%BE%B5%E8%A2%AD%E6%8C%81%E7%BB%AD%E4%B8%AD%20-%20HDU%203065.jpg)
**题目大意**  
（继续略！）   
<br>
**思路**    
仍然算是模板题    
记得搜到时就++该点记录的pos的cnt值  
```cpp
  #include <cstring>
  #include <cstdio>
  #include <algorithm>
  using namespace std;
  const int N = 5e4 + 15;
  const int ascii_size = 96;

  char in[1005][55];
  int  cnt[N];
  bool pend[N];
  int  nxt[N][ascii_size];
  int  fail[N];
  int  pos[N];
  int  tot;

  int  que[N];
  int  head, tail;

  inline void init(){
      memset(cnt, 0, sizeof(cnt));
      memset(pos, 0, sizeof(pos));
      memset(nxt[0], -1, sizeof(nxt));
      memset(fail, 0, sizeof(fail));
      head = tail = 0;
      tot = 1;
  }

  void push(char* s, int i){
      int p = 0;
      for(; *s; s++){
          int idx = * s - 32;
          if(nxt[p][idx] == -1){
              nxt[p][idx] = tot++;
          }
          p = nxt[p][idx];
      }
      pos[p] = i;
  }

  void setFail(){
      for(int idx = 0; idx < ascii_size; idx++){
          if(~nxt[0][idx]){
              que[head++] = nxt[0][idx];
          }else{
              nxt[0][idx] = 0;
          }
      }
      while(tail != head){
          int p = que[tail++];
          for(int idx = 0; idx < ascii_size; idx++){
              if(~nxt[p][idx]){
                  fail[nxt[p][idx]] = nxt[fail[p]][idx];
                  que[head++] = nxt[p][idx];
              }else{
                  nxt[p][idx] = nxt[fail[p]][idx];
              }
          }
      }
  }

  void query(){
      int p = 0;
      char ch;
      while((ch = getchar()) != '\n'){
          int idx = ch - 32;
          p = nxt[p][idx];
          for(int q = p; pos[q] && q; q = fail[q]){
              cnt[pos[p]]++;
          }
      }
  }

  int main(){
      int n;
      while(~scanf("%d", &n)){
          init();
          for(int i = 1; i <= n; i++){
              scanf("%s", in[i]);
              push(in[i], i);
          }
          getchar();
          setFail();

          query();
          for(int i = 1; i <= n; i++){
              if(cnt[i])  printf("%s: %d\n", in[i], cnt[i]);
          }

      }
      return 0;
  }
```

### DNA Sequence - POJ 2778
![](http://houzajblog-1252277898.coscd.myqcloud.com/20180420%20Problem0420/DNA%20Sequence%20-%20POJ%202778.jpg)
**题目大意**  
给定一些病毒串，问长度为n且不包含病毒串的DNA序列有多少种，结果对100000取模       
<br>
**思路**    
这题用到了离散数学的一个定理，
> 如果邻接矩阵中c(i, j)代表从节点i走到节点j走一步的不同走法，那么矩阵的k次方中的c'(i, j)代表从节点i走到节点j走k步的不同走法  

所以这题可用AC自动机构建一个邻接矩阵，这个矩阵是排除了走向病毒串的矩阵，然后再进行矩阵快速幂，最后答案就是 sigma\[i](c\[0]\[i])  
```cpp
  #include <cstdio>
  #include <cstring>
  #include <iostream>
  #include <algorithm>
  using namespace std;
  typedef long long ll;
  const int N = 4;
  const int M = 100 + 5;
  const int MOD = 100000;

  ll base[M][M];
  ll tmp[M][M];

  int  nxt[M][N];
  bool mark[M];
  int  fail[M];
  int  tot;

  int  que[M];
  int  head, tail;

  ll  a[M][M];
  ll  ret[M][M];

  struct Matrix{
      void multiply(ll a[M][M], ll b[M][M], ll ret[M][M]){
          for(int i = 0; i < tot; i++){
              for(int j = 0; j < tot; j++){
                  ll sum = 0;
                  for(int k = 0; k < tot; k++){
                      sum = (sum + a[i][k] * b[k][j]);
                  }
                  tmp[i][j] = sum%MOD;    //先计算再求模，不然会TLE，这也说明了求模的确很耗时
              }
          }
          mcopy(ret, tmp);
      }

      void mcopy(ll d[M][M], ll s[M][M]){
          for(int i = 0; i < M; i++){
              for(int j = 0; j < M; j++){
                  d[i][j] = s[i][j];
              }
          }
      }

      void quickPow(ll a[M][M], ll b, ll ret[M][M]){
          mcopy(base, a);
          memset(ret[0], 0, 4*M*M);
          for(int i = 0; i < M; i++)      ret[i][i] = 1;

          while(b){
              if(b&1)     multiply(ret, base, ret);
              multiply(base, base, base);
              b >>= 1;
          }
      }
  };

  inline void init(){
      memset(nxt[0], -1, sizeof(nxt));
      memset(mark, 1, sizeof(mark));
      memset(fail, 0, sizeof(fail));
      tot = 1;
      head = tail = 0;
  }

  int cvt(char ch){
      if(ch == 'A')   return 0;
      if(ch == 'T')   return 1;
      if(ch == 'C')   return 2;
      else            return 3;
  }

  void build(char* s){
      int p = 0;
      while(*s){
          int idx = cvt(*s);
          if(nxt[p][idx] == -1){
              nxt[p][idx] = tot++;
          }
          mark[nxt[p][idx]] &= mark[p];   //从病毒串走出来的，就需要更新把mark值设为0
          p = nxt[p][idx];
          s++;
      }
      mark[p] = false;
  }

  void autoMaton(){
      for(int idx = 0; idx < N; idx++){
          if(nxt[0][idx] != -1){
              que[head++] = nxt[0][idx];
          }else{
              nxt[0][idx] = 0;
          }
      }
      while(tail != head){
          int p = que[tail++];
          //Fail指针指向的点是病毒串，那么该点也是病毒串的点
          //因为Fail指针指向的点一定是该串的一个后缀
          mark[p] &= mark[fail[p]];   
          for(int idx = 0; idx < N; idx++){
              if(nxt[p][idx] == -1){
                  nxt[p][idx] = nxt[fail[p]][idx];
              }else{
                  fail[nxt[p][idx]] = nxt[fail[p]][idx];
                  que[head++] = nxt[p][idx];
              }
          }
      }
  }

  int res(ll b){
      ll ans = 0;
      Matrix mat;
      memset(a, 0, sizeof(a));
      for(int i = 0; i < tot; i++){
          for(int j = 0; j < N; j++){
              a[i][nxt[i][j]] += (mark[i] & mark[nxt[i][j]]);   //排除掉病毒串的点
          }
      }
      mat.quickPow(a, b, ret);
      for(int i = 0; i < tot; i++){
          ans += ret[0][i];
      }
      return ans%MOD;
  }

  int main(){
      int n;
      ll b;
      char str[15];
      while(~scanf("%d%lld", &n, &b)){
          init();
          while(n--){
              scanf("%s", str);
              build(str);
          }
          autoMaton();
          printf("%d\n", res(b));
      }
      return 0;
  }
```
